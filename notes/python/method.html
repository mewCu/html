函数定义

def clac_add(a,b):
    return a+b
def fun1(a,b,c=10): # c的默认值是10,fun1(1,2)返回值是13
    return a+b+c
def fun2(list_src): #把列表中的数分成偶数和奇数2个列表并返回该2个列表
    odd = []
    even = []
    for i in list_src:
        if i%2 == 0:
            odd.append(i)
        else:
            even.append(i)
return odd, even
def fun3(*args): # 可变参数fun3(a1,a2,a3,an)
    pass # args其实是元组
def fun4(**args): # 形参可变参数fun4(b=1,a=2,f=4)
    pass #args其实是字典
def fun5(*a, **bb): # 不能写成fun5(**aa, *b),fun5(*a, *b)或fun5(**aa, **bb)
    pass
def fun6(a,b,*,c,d): # *之后形参必须指定,如fun6(1,2,d=4,c=3)
    pass
def fun7(a,b,*,c,d,**ee): # fun7(1,2,d=4,c=3,e=5) 从第5个参数开始会以字典的形式传入ee
    print(ee) # {'e':5}
def fun8(a,b,*,*c): # 会报错,*后面不能接元组
