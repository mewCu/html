class类的创建

class Dog:

    name = 'xiaogougou'

    def __init__(self,name,color):
        self.name=name
        self.color=color

    def eat(self,food): # 实例方法 
        print('%s is eating %s' % (self.name, food))

    @staticmethod # 静态方法
    def sleep():
        print('z Z')

    @classmethod # 类方法
    def cm(cls):
        print('classmethod')


dog = Dog('xiaohei','wihte')
dog.eat('bone')
dog.sleep()
Dog.sleep() 
Dog.cm() # 类调用类方法,工具类写法
Dog.eat(dog,'apple') # xiaohei is eating apple
Dog.eat(Dog,'apple') # xiaogougou is eating apple
print(dog.color) # wihte
print(Dog.color) # 报错
